One of my favorite things I learned in this class was the different ways to write a sort algorithm. Since most languages have built in sort functions, I never realized there are various ways to get a sorted list. This class introduced me to two sort algorithms: the insertion sort and the selection sort. For fun, I wrote my own selection sort function in Xcode's playground using Swift:


    // Selection Sort Function
    func selectionSort(var arr: [Int]) -> [Int] {

        var ln, lni, cv: Int;
        for(var i = 0; i < arr.count; i++) {
            cv = arr[i];
            ln = arr[i];
            lni = i;
            
            for(var j = i + 1; j < arr.count; j++) {
                if(arr[j] < arr[lni]) {
                    ln = arr[j];
                    lni = j;
                }
            }
            
            arr[i] = ln;
            arr[lni] = cv;
        }
        
        return arr;
    }

    selectionSort([4,3,56,2,5]);
    // prints [2, 3, 4, 5, 56]
